排序

```python
n=int(input())
d={}
for _ in range(n):
    name,num=input().split('-')
    if name in d:
        d[name].append(num)
    else:
        d[name]=[num]
sorted_name=sorted(d.keys())
for name in sorted_name:
    l1,l2=[],[]
    for num in d[name]:
        if num[-1]=='M':
            l1.append(num)
        elif num[-1]=='B':
            l2.append(num)
    l1.sort(key=lambda x:float(x[:-1]))
    l2.sort(key=lambda x:float(x[:-1]))
    l=l1+l2
    print(f'{name}:',end=' ')
    print(', '.join(l))
```
