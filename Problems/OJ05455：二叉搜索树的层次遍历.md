构建二叉搜索树：递归思想。如果数字比当前节点小，就递归插入左子树，如果大就递归插入右子树，如果相等就直接返回当前节点（重复数字只记入一个）。

层序遍历：使用双端队列的bfs。

```python
class TreeNode:
    def __init__(self,val=0,left=None,right=None):
        self.val=val
        self.left=left
        self.right=right
def insert(node,num):
    if not node:
        return TreeNode(num)
    if node.val>num:
        node.left=insert(node.left,num)
    elif node.val<num:
        node.right=insert(node.right,num)
    return node
from collections import deque
def traversal(tree):
    queue=deque([tree])
    while queue:
        node=queue.popleft()
        ans.append(node.val)
        if node.left:
            queue.append(node.left)
        if node.right:
            queue.append(node.right)
nums=list(map(int,input().split()))
root=None
for n in nums:
    root=insert(root,n)
ans=[]
traversal(root)
print(*ans)
```
