利用哨兵节点和环形链表，定义增添和删除两个辅助函数。所有的增添操作都在哨兵节点的next，删除操作在任意位置。

当执行get时，需要将被提取的节点从原位置删除，并重新增添，这样就实现了节点使用后提到最前。

当执行put时，如果超出容量，哨兵节点的last即为最久未使用的，将其删除，并删除字典中对应的键

```python
class Node:
    def __init__(self,val=0,key=0,last=None,next=None):
        self.val=val
        self.key=key
        self.last=last
        self.next=next
class LRUCache:

    def __init__(self, capacity: int):
        self.c=capacity
        self.sen=Node()
        self.sen.last=self.sen
        self.sen.next=self.sen
        self.d={}
    def delete(self,node):
        node.next.last=node.last
        node.last.next=node.next
    def add(self,node):
        nxt=self.sen.next
        self.sen.next=node
        node.next=nxt
        node.last=self.sen
        nxt.last=node
    def get(self, key: int) -> int:
        if key in self.d:
            node=self.d[key]
            self.delete(node)
            self.add(node)
            return node.val
        return -1

    def put(self, key: int, value: int) -> None:
        if key in self.d:
            node=self.d[key]
            self.delete(node)
            self.add(node)
            node.val=value
            return
        self.d[key]=Node(value,key)
        self.add(self.d[key])
        if len(self.d)>self.c:
            del_node=self.sen.last
            self.delete(del_node)
            del self.d[del_node.key]
```
