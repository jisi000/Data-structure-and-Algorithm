按题目描述建队，处理数据即可。

```python
n=int(input())
io=list(input().split())
lo=[[] for _ in range(9)]
for ch in io:
    lo[int(ch[1])-1].append(ch)
it=[]
for i in range(9):
    it.extend(lo[i])
    print(f'Queue{i+1}:',end='')
    print(' '.join(lo[i]))
lt=[[] for _ in range(4)]
ans=[]
for ch in it:
    lt[ord(ch[0])-65].append(ch)
for i in range(4):
    ans.extend(lt[i])
    print(f'Queue{chr(i+65)}:',end='')
    print(' '.join(lt[i]))
print(' '.join(ans))
```
