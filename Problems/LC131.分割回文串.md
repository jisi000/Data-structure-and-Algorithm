先用二维动态规划记录字符串中的回文子串，再用回溯得到分割方法

动态规划采用左闭右开，is_parl[i] [j]由is_parl[i+1] [j-1]得到

```python
class Solution:
    def partition(self, s: str) -> List[List[str]]:
        n=len(s)
        is_parl=[[False]*(n+1) for _ in range(n+1)]
        for i in range(n+1):
            is_parl[i][i]=True
            if i<n:
                is_parl[i][i+1]=True
        for start in range(n-1,-1,-1):
            for end in range(start+2,n+1):
                if s[start]==s[end-1] and is_parl[start+1][end-1]:
                    is_parl[start][end]=True
        ans=[]
        re=[]
        def backtrack(start):
            if start==n:
                ans.append(re[:])
                return
            for end in range(start+1,n+1):
                if is_parl[start][end]:
                    re.append(s[start:end])
                    backtrack(end)
                    re.pop()
        backtrack(0)
        return ans
```
