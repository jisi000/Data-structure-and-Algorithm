使用dfs。为了降低时间复杂度，没有使用visited列表记录访问过的位置，而是把访问过的位置原地修改为''，再恢复现场。

```python
class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        r,c=len(board),len(board[0])
        d=[(1,0),(0,1),(-1,0),(0,-1)]
        def dfs(lo,wordi):
            if not wordi:
                return True
            re=wordi[0]
            for dx,dy in d:
                x,y=lo[0]+dx,lo[1]+dy
                if 0<=x<r and 0<=y<c and board[x][y]==re:
                    board[x][y]=''
                    if dfs((x,y),wordi[1:]):
                        return True
                    board[x][y]=re
            return False
        for i in range(r):
            for j in range(c):
                if board[i][j]==word[0]:
                    board[i][j]=''
                    if dfs((i,j),word[1:]):
                        return True
                    board[i][j]=word[0]
        return False
```
