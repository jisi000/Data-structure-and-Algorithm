如果第i行不满足条件，找到其下第一个满足条件的行，把它翻到最上面，翻转次数是j-i。

```python
class Solution:
    def minSwaps(self, grid: List[List[int]]) -> int:
        ans=0
        n=len(grid)
        pos=[-1]*n
        for i in range(n):
            for j in range(n-1,-1,-1):
                if grid[i][j]==1:
                    pos[i]=j
                    break
        for i in range(n-1):
            if pos[i]<=i:
                continue
            for j in range(i+1,n):
                if pos[j]<=i:
                    ans+=j-i
                    for x in range(j,i,-1):
                        pos[x],pos[x-1]=pos[x-1],pos[x]
                    break
            if pos[i]>i:
                return -1
        return ans
```
